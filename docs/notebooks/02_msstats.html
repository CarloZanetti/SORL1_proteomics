Build annotation from study design

All 8 samples passed QC. R.Condition in the raw file is “Not Defined” — we assign WT/KO here from the known sample table.

Note: WT samples are all Male (BDAG171 cohort); KO samples are 2 Female + 2 Male (BDAG173 cohort). Sex is not modelled as a covariate in the MSstats linear model — the KO vs WT contrast is therefore partially confounded with sex.

Feature completeness filter

Keep features observed in ≥50% of runs within at least one condition (≥ 2 of 4 samples in WT or ≥ 2 of 4 samples in KO).

Code 
# Manual resume — skip rebuild on re-run
msstats_input <- qs_read(file.path(objects_dir, "msstats_input.qs2"))
Code 
msstats_input_filt <- as.data.frame(msstats_input) %>%
  group_by(ProteinName, PeptideSequence, PrecursorCharge,
           FragmentIon, ProductCharge, Condition) %>%
  summarise(
    n_obs   = sum(!is.na(Intensity) & Intensity > 0),
    n_total = n(),
    prop    = n_obs / n_total,
    .groups = "drop"
  ) %>%
  group_by(ProteinName, PeptideSequence, PrecursorCharge,
           FragmentIon, ProductCharge) %>%
  filter(any(prop >= 0.5)) %>%
  ungroup() %>%
  inner_join(
    as.data.frame(msstats_input),
    by = c("ProteinName", "PeptideSequence", "PrecursorCharge",
           "FragmentIon", "ProductCharge", "Condition")
  )

Build contrast matrix

Single comparison: KO vs WT. Positive log2FC = higher abundance in KO.

Code 
make_contrast <- function(numerator, denominator, levels) {
  v <- setNames(rep(0L, length(levels)), levels)
  v[numerator]   <-  1L
  v[denominator] <- -1L
  v
}

comparison_matrix <- rbind(
  make_contrast("KO", "WT", condition_levels)
)
rownames(comparison_matrix) <- "KO_vs_WT"

msstats_input_filt <- as.data.frame(msstats_input) %>%
  group_by(ProteinName, PeptideSequence, PrecursorCharge,
           FragmentIon, ProductCharge, Condition) %>%
  summarise(
    n_obs   = sum(!is.na(Intensity) & Intensity > 0),
    n_total = n(),
    prop    = n_obs / n_total,
    .groups = "drop"
  ) %>%
  group_by(ProteinName, PeptideSequence, PrecursorCharge,
           FragmentIon, ProductCharge) %>%
  filter(any(prop >= 0.5)) %>%
  ungroup() %>%
  inner_join(
    as.data.frame(msstats_input),
    by = c("ProteinName", "PeptideSequence", "PrecursorCharge",
           "FragmentIon", "ProductCharge", "Condition")
  )

message("Rows before completeness filter: ", nrow(msstats_input))

## Build contrast matrix

Single comparison: KO vs WT. Positive log2FC = higher abundance in KO.

```{r}
make_contrast <- function(numerator, denominator, levels) {
  v <- setNames(rep(0L, length(levels)), levels)
  v[numerator]   <-  1L
  v[denominator] <- -1L
  v
}

comparison_matrix <- rbind(
  make_contrast("KO", "WT", condition_levels)
)
rownames(comparison_matrix) <- "KO_vs_WT"

print(comparison_matrix)
```
