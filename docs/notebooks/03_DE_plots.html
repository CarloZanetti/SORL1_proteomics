Volcano plot function

FDR threshold: 0.05. Fold-change threshold: ±0.5 log2FC (configurable). Top 20 significant genes per direction are labelled, ranked by FDR.

Code 
volcano_cols <- c(
  "Upregulated"     = "firebrick3",
  "Downregulated"   = "navy",
  "Not Significant" = "grey80"
)

generate_swanky_volcano <- function(msstats_results, dict, comparison_label,
                                    logfc_cutoff = 0.5, graphs_dir = "results") {
  volcano_data <- msstats_results %>%
    filter(Label == comparison_label, !is.na(adj.pvalue), is.finite(log2FC)) %>%
    left_join(dict, by = "Protein") %>%
    mutate(
      Gene   = ifelse(is.na(Gene) | Gene == "", Protein, Gene),
      Status = case_when(
        adj.pvalue < 0.05 & log2FC >  logfc_cutoff ~ "Upregulated",
        adj.pvalue < 0.05 & log2FC < -logfc_cutoff ~ "Downregulated",
        TRUE                                        ~ "Not Significant"
      )
    ) %>%
    group_by(Status) %>%
    mutate(GroupRank = rank(adj.pvalue, ties.method = "first")) %>%
    ungroup() %>%
    mutate(
      PlotLabel = case_when(
        Status != "Not Significant" & GroupRank <= 20 ~ Gene,
        TRUE                                          ~ NA_character_
      )
    )

  p <- ggplot(volcano_data,
              aes(x = log2FC, y = -log10(adj.pvalue),
                  fill = Status, color = Status)) +
    geom_point(aes(size = Status, alpha = Status), shape = 21, stroke = 0.2) +
    scale_fill_manual(values = volcano_cols) +
    scale_color_manual(values = c(
      "Upregulated"     = "black",
      "Downregulated"   = "black",
      "Not Significant" = "grey90"
    )) +
    scale_size_manual(values = c(
      "Upregulated"     = 3,
      "Downregulated"   = 3,
      "Not Significant" = 1.5
    )) +
    scale_alpha_manual(values = c(
      "Upregulated"     = 1,
      "Downregulated"   = 1,
      "Not Significant" = 0.3
    )) +
    geom_hline(yintercept = -log10(0.05),
               linetype = "dashed", color = "grey30", linewidth = 0.8) +
    geom_vline(xintercept = c(-logfc_cutoff, logfc_cutoff),
               linetype = "dashed", color = "grey30", linewidth = 0.8) +
    geom_label_repel(
      aes(label = PlotLabel),
      fill = "white", color = "black", fontface = "bold", size = 3.5,
      box.padding = 0.5, point.padding = 0.5,
      min.segment.length = 0, segment.color = "grey50", max.overlaps = 50
    ) +
    labs(
      title    = paste0("Differential Abundance: ", gsub("_", " ", comparison_label)),
      subtitle = paste0("Top 20 Genes Labeled | FDR < 0.05 | |log2FC| > ", logfc_cutoff),
      x        = bquote(bold("Log"[2] ~ "Fold Change")),
      y        = bquote(bold("-Log"[10] ~ "FDR"))
    ) +
    theme_minimal(base_size = 14) +
    theme(
      plot.title       = element_text(face = "bold", size = 18, hjust = 0),
      plot.subtitle    = element_text(size = 12, color = "grey40",
                                      margin = margin(b = 10)),
      axis.title       = element_text(face = "bold", size = 14),
      axis.text        = element_text(face = "bold", color = "black"),
      axis.line        = element_line(color = "black", linewidth = 1),
      axis.ticks       = element_line(color = "black", linewidth = 1),
      legend.position  = "top",
      legend.title     = element_blank(),
      legend.text      = element_text(size = 12, face = "bold"),
      panel.grid.minor = element_blank(),
      panel.grid.major = element_line(color = "grey95")
    )

  dir.create(graphs_dir, showWarnings = FALSE, recursive = TRUE)
  safe_name <- gsub(" ", "_", comparison_label)

  ggsave(
    filename = file.path(graphs_dir, paste0("Volcano_", safe_name, ".pdf")),
    plot     = p,
    width    = 10, height = 7, dpi = 300,
    device   = cairo_pdf
  )

  export_data <- volcano_data %>%
    dplyr::select(Protein, Gene, Description, log2FC, pvalue, adj.pvalue, Status) %>%
    arrange(adj.pvalue)

  fwrite(export_data,
         file = file.path(graphs_dir, paste0("Volcano_Data_", safe_name, ".csv")))

  return(p)
}

One-condition-missing proteins

Proteins with issue == "oneConditionMissing" are detected only in WT or only in KO. MSstats assigns them log2FC = ±Inf and excludes them from the volcano (no finite x-coordinate). They are reported here as a separate presence/absence table.

The number of samples in which the protein was detected (from the cleaned pre-imputation Spectronaut data) is shown so genuine on/off candidates (≥3 of 4 samples in the present condition) can be separated from sparse detections that likely reflect noise.

Code 
clean_meta <- fread(file.path(objects_dir, "clean_metadata.csv"))

# Detection count per protein per genotype, from the cleaned (pre-imputation)
# fragment-level data. PG.Quantity is the same for every fragment of a given
# (sample, protein) pair, so a protein is "detected" in a sample if any row
# has non-NA PG.Quantity.
detection_counts <- clean_spec %>%
  filter(!is.na(PG.Quantity)) %>%
  distinct(PG.ProteinGroups, R.FileName) %>%
  left_join(
    clean_meta %>% dplyr::select(FileName, Genotype),
    by = c("R.FileName" = "FileName")
  ) %>%
  group_by(PG.ProteinGroups) %>%
  summarise(
    n_WT = sum(Genotype == "WT"),
    n_KO = sum(Genotype == "KO"),
    .groups = "drop"
  )

one_cond <- res_df %>%
  filter(issue == "oneConditionMissing") %>%
  left_join(protein_dictionary, by = "Protein") %>%
  left_join(detection_counts, by = c("Protein" = "PG.ProteinGroups")) %>%
  mutate(
    n_WT      = ifelse(is.na(n_WT), 0L, n_WT),
    n_KO      = ifelse(is.na(n_KO), 0L, n_KO),
    Direction = case_when(
      log2FC ==  Inf ~ "KO-only",
      log2FC == -Inf ~ "WT-only",
      TRUE           ~ NA_character_
    ),
    n_present  = pmax(n_WT, n_KO),
    Confidence = case_when(
      n_present == 4 ~ "High (4/4)",
      n_present == 3 ~ "Moderate (3/4)",
      n_present == 2 ~ "Low (2/4)",
      n_present == 1 ~ "Very low (1/4)",
      TRUE           ~ "None"
    )
  ) %>%
  arrange(desc(n_present), Direction, Gene) %>%
  dplyr::select(Protein, Gene, Description, Direction,
                n_WT, n_KO, n_present, Confidence)

fwrite(one_cond, file.path(results_dir, "OneConditionMissing_KO_vs_WT.csv"))
